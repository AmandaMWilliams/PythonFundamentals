#### Indentation

if condition:

<br>
<hr>

x = 42  
if x > 5:  
&nbsp;&nbsp;&nbsp;&nbsp; print("x is greater than 5")

<hr>

num = 12  
if num > 5:  
&nbsp;&nbsp;&nbsp;&nbsp; print("Bigger than 5")  
&nbsp;&nbsp;&nbsp;&nbsp; if num <= 47:  
&nbsp;&nbsp;&nbsp;&nbsp; &nbsp;&nbsp;&nbsp;&nbsp; print("Between 5 and 47")
